# -----------------------------------------------------------------------------
# Streamlit 프론트엔드 화면 구성 패키지
# -----------------------------------------------------------------------------

# Python 코드만으로 웹 화면을 빠르게 만들기 위한 UI 프레임워크입니다.
# 이 과정의 중심 프론트엔드 도구입니다.
# 사용 위치:
# - 01_streamlit-basic: Streamlit 실행, 텍스트, 버튼, 입력 컴포넌트 기초
# - 02_streamlit-ui-components: 입력폼, 선택박스, 파일 업로드, 표와 차트 화면 구성
# - 05_ai-chatbot-interface: 챗봇 입력 화면과 응답 표시 화면
# - 04_state-session-and-data: st.session_state 기반 대화 상태 관리
# - 06_streamlit-multipage-app: st.Page/st.navigation 기반 왼쪽 메뉴 구조
# - 07_streamlit-tabs-app: st.tabs 기반 탭 구조, tab_pages 파일 분리
# - 90_ai-assisted-ui-review-and-debugging: Streamlit UI 오류 점검
# - 99_final-frontend-project: 최종 프론트엔드 solution과 제출물 작성
streamlit==1.41.1

# 표, 데이터프레임, 차트 표시 패키지

# 사용 위치:
# - 02_streamlit-ui-components: 표와 차트 표시
# - 04_state-session-and-data: 대화 이력과 사용자 상태 데이터 표시
# - 06_streamlit-multipage-app: mock 로그와 mock 데이터베이스 테이블 표시
# - 07_streamlit-tabs-app: 탭 화면 안의 mock 로그와 mock 데이터베이스 테이블 표시
# - 99_final-frontend-project: 백엔드 응답을 표로 표시하는 최종 화면
pandas==2.2.3


# -----------------------------------------------------------------------------
# FastAPI 백엔드 호출 패키지

# - 04_state-session-and-data: 사용자별 대화 이력, 서비스 로그 API 호출 흐름 이해
# - 99_final-frontend-project: backend_mock 또는 backend_service의 /health, /auth, /chat, /conversations, /service-logs 호출

python-dotenv==1.0.1
